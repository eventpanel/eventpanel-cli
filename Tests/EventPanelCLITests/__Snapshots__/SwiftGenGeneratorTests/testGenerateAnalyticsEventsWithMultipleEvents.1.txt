import Foundation

internal enum AnalyticsEvents {
  internal enum Onboarding {

    /// User started the onboarding process
    /// - Parameters:
    ///     - source: Where the onboarding was initiated from
    internal static func onboardingStart(  
      source: String
    ) -> AnalyticsEvent {
      AnalyticsEvent(
        name: "Onboarding Start",
        parameters: [
          "source": source
        ]
      )
    }

    /// User completed the onboarding process
    /// - Parameters:
    ///     - completion_time: Time taken to complete onboarding in seconds
    ///     - skipped_steps: Number of steps skipped during onboarding
    internal static func onboardingComplete(  
      completionTime: Int,
      skippedSteps: Int?
    ) -> AnalyticsEvent {
      AnalyticsEvent(
        name: "Onboarding Complete",
        parameters: [
          "completion_time": completionTime,
          "skipped_steps": skippedSteps
        ].byExcludingNilValues()
      )
    }
  }
}
